Abella < Kind nat type.


Abella < Type z nat.


Abella < Type s nat -> nat.


Abella < Define nat : nat -> prop by 
nat z;
nat (s N) := nat N.


Abella < Define even : nat -> prop by 
even z;
even (s (s N)) := even N.


Abella < Define odd : nat -> prop by 
odd (s z);
odd (s (s N)) := odd N.
