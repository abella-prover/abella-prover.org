Abella < Type is_o o -> prop.


Abella < Define $is_list : nat -> (list o) -> prop by 
$is_list (s X) nil := is_nat X;
$is_list (s X) (A :: L) := is_o A /\ $is_list X L.


Abella < Define is_list : (list o) -> prop by 
is_list L := exists X, is_nat X /\ $is_list X L.


Abella < Define $adj : nat -> (list o) -> o -> (list o) -> prop by 
$adj (s X) L A (A :: L) := is_nat X /\ is_o A /\ is_list L;
$adj (s X) (B :: K) A (B :: L) := is_o B /\ $adj X K A L.


Abella < Define adj : (list o) -> o -> (list o) -> prop by 
adj K A L := exists X, is_nat X /\ $adj X K A L.


Abella < Define $perm : nat -> (list o) -> (list o) -> prop by 
$perm (s X) nil nil := is_nat X;
$perm (s X) K L := exists A KK LL, adj KK A K /\ adj LL A L /\ $perm X KK LL.


Abella < Define perm : (list o) -> (list o) -> prop by 
perm K L := exists X, is_nat X /\ $perm X K L.
