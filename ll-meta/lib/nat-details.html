Abella < Kind nat Type.


Abella < Type z nat.


Abella < Type s nat -> nat.


Abella < Define is_nat : nat -> prop by 
is_nat z;
is_nat (s X) := is_nat X.


Abella < Close nat.


Abella < Define plus : nat -> nat -> nat -> prop by 
plus z X X := is_nat X;
plus (s X) Y (s Z) := plus X Y Z.


Abella < Define leq : nat -> nat -> prop by 
leq z X := is_nat X;
leq (s X) (s Y) := leq X Y.
