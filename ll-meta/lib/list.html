Type is_o  o -> prop.

Define is_list : olist -> prop by
; is_list nil
; is_list (A :: L) := is_o A /\ is_list L
.

Define append : olist -> olist -> olist -> prop by
; append nil L L := is_list L
; append (A :: J) K (A :: L) := is_o A /\ append J K L
